git clone "Link do repositorio" = pega o repositorio do github e transfere pra sua maquina
git chekout -b "nome-repositorio" cria e já entra no repositorio criado 
git status = verifica status atual do repositorio
git add --all = adiciona todos os arquivos modificados 
git add . = adiciona todos os arquivos dentro da pasta
git log = mostra historico do repositorio
git commit -m "mensagem-a-ser-exibida" = manda o repositorio local para o github
git init = cria repositorio local 
git branch = cria ramificação do projeto
git push origin 'nome-da-branch' = envia para sua conta do github a branch desejada
